# installing Modules
import requests
from bs4 import BeautifulSoup
import csv
import pandas as p
# scraping information
url = "https://www.flipkart.com/search?q=mobiles"
r = requests.get(url)
soup = BeautifulSoup(r.content,"html.parser")
titles = soup.find_all('div',class_='_4rR01T')
ratings = soup.find_all('div',class_='_3LWZIK')
reviews = soup.find_all('span',class_='_2_R_DZ')
prices = soup.find_all('div',class_='_30jeq3 _1_WHN1')
mt = []
mra = []
mre = []
mpr = []
for tit,rat,rew,pri in zip(titles,ratings,reviews,prices):
    mt.append(tit.text)
    mra.append(rat.text)
    mre.append(rew.text)
    mpr.append(pri.text)
# saving data in csv
d = {'mt':mt,'mra':mra,'mre':mre,'mpr':mpr}
model = p.DataFrame(data=d)

model.to_csv("Mobiles.csv")
